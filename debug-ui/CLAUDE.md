# Client Agent Instructions

This is the web debug monitor for the Stripe Agentic City. It connects to the game server via WebSocket and renders the world state using FlatBuffers binary deserialization.

## Architecture

```
Game Server (Bevy ECS)
    │
    │ FlatBuffers binary (WorldSnapshot)
    │ via WebSocket at ws://localhost:8080/ws
    ▼
Web Client (TypeScript + Vite)
    │
    ├── src/main.ts          — main render loop, WebSocket connection
    ├── src/generated/       — auto-generated FlatBuffers TypeScript
    ├── index.html           — UI layout + CSS
    └── package.json         — dependencies (flatbuffers)
```

## FlatBuffers Schema

The schema is defined in `schemas/schemas/world.fbs`. TypeScript bindings are auto-generated by `flatc --ts` during the schema build (`cargo build -p schemas`).

### Root Type: `WorldSnapshot`

Every WebSocket message is a binary FlatBuffer containing:

```
WorldSnapshot {
  tick: uint64                      — current game tick
  agents: [AgentSnapshot]           — all agent states
  structures: [StructureSnapshot]   — all building states
  bounties: [BountySnapshot]        — bounty board
  board_queue: BoardQueueSnapshot   — who's at the board
  event_log: [LogEntrySnapshot]     — activity log entries
}
```

### AgentSnapshot Fields

```typescript
const agent = snapshot.agents(i)!;

agent.id()           // UUID string
agent.name()         // "Alice", "Bob", "Carol"
agent.pos()          // Vec2i — .x(), .y()
agent.animation()    // AnimState enum: Idle=0, Walking=1, Working=2
agent.thought()      // current thought bubble text (full, not truncated)
agent.gold()         // gold coin count
agent.goal()         // debug string like "ExecutingBounty(uuid)" or "Idle"
agent.inventoryLength()  // number of inventory slots
agent.inventory(j)       // InventorySlot — .itemType(), .count()
                         // Special prefixes:
                         //   "card:Bob"              — business card
                         //   "doc:research.md"       — named document (clickable)
                         //   "bounty: Warehouse..." — active bounty notice

// Needs (struct — inline, not nullable)
const needs = agent.needs()!;
needs.energy()   // 0-100, driven by token usage
needs.hunger()   // 0-100, decays over time
needs.boredom()  // 0-100, decays when idle

// Token tracking stats (table — nullable)
const stats = agent.stats();
if (stats) {
  stats.tokensUsed()      // cumulative tokens this session
  stats.contextLimit()    // token budget (default 150k)
  stats.totalCostUsd()    // lifetime USD cost
  stats.sessionCostUsd()  // current session USD cost
}

// Relationships
for (let j = 0; j < agent.relationshipsLength(); j++) {
  const rel = agent.relationships(j)!;
  rel.agentName()    // who they know
  rel.friendship()   // friendship points
  rel.lastGoal()     // last known goal of that agent
}

// Active conversation messages
for (let j = 0; j < agent.activeChatLength(); j++) {
  const msg = agent.activeChat(j)!;
  msg.speaker()   // who said it
  msg.text()      // what they said
}

// Visible entities (from look_around)
for (let j = 0; j < agent.visibleEntitiesLength(); j++) {
  const v = agent.visibleEntities(j)!;
  v.name()      // entity name
  v.pos()       // Vec2i position
  v.kind()      // "Agent" or "Structure"
  v.distance()  // manhattan distance
}
```

### StructureSnapshot

```typescript
const structure = snapshot.structures(i)!;
structure.id()           // UUID
structure.spriteType()   // "cafe", "hotel", "warehouse", etc.
structure.pos()          // Vec2i
structure.interactable() // bool
// Inventory
for (let j = 0; j < structure.inventoryLength(); j++) {
  const slot = structure.inventory(j)!;
  slot.itemType()  // "coffee_beans", "gold_egg", etc.
  slot.count()     // quantity
}
```

### BountySnapshot

```typescript
const bounty = snapshot.bounties(i)!;
bounty.id()          // UUID
bounty.description() // title text
bounty.rewardGold()  // gold reward
bounty.state()       // BountyStatus enum:
                     //   0 = Available, 1 = Claimed,
                     //   2 = Completed, 3 = Submitted (pending GM review)
bounty.claimedBy()   // agent name or null
```

### LogEntrySnapshot

```typescript
const entry = snapshot.eventLog(i)!;
entry.tick()   // game tick when event occurred
entry.agent()  // agent name (or "SYSTEM" for GM commentary)
entry.kind()   // "thought", "action", "speech", "system", "decision"
entry.text()   // event content
```

## How to Consume the WebSocket

```typescript
import * as flatbuffers from "flatbuffers";
import { WorldSnapshot } from "./generated/world/world-snapshot.js";

const ws = new WebSocket("ws://localhost:8080/ws");
ws.binaryType = "arraybuffer";

ws.onmessage = (event) => {
  const buf = new flatbuffers.ByteBuffer(new Uint8Array(event.data));
  const snapshot = WorldSnapshot.getRootAsWorldSnapshot(buf);

  // Access tick
  console.log("Tick:", snapshot.tick());

  // Iterate agents
  for (let i = 0; i < snapshot.agentsLength(); i++) {
    const agent = snapshot.agents(i)!;
    console.log(agent.name(), agent.gold(), "gold");
  }
};
```

## Updating the Render

The render function receives a `WorldSnapshot` every tick (~2Hz). The pattern:

1. **Parse the snapshot** — iterate agents, structures, bounties, logs
2. **Build HTML strings** — create the UI elements
3. **Set innerHTML** — update the DOM

For persistent state (like thought history), maintain a client-side cache keyed by agent name and append new entries each frame.

```typescript
// Per-agent thought history (persists across renders)
const agentThoughts: Record<string, string[]> = {};

function render(s: WorldSnapshot) {
  // Collect new thoughts from event log
  for (let i = 0; i < s.eventLogLength(); i++) {
    const entry = s.eventLog(i)!;
    if (entry.kind() === "thought") {
      const agent = entry.agent()!;
      if (!agentThoughts[agent]) agentThoughts[agent] = [];
      agentThoughts[agent].push(entry.text()!);
    }
  }

  // Render agent cards
  for (let i = 0; i < s.agentsLength(); i++) {
    const agent = s.agents(i)!;
    const name = agent.name()!;
    const thoughts = agentThoughts[name] || [];
    // ... build card HTML with latest thoughts
  }
}
```
